Overview

simple_htmx is a fluent HTML and HTMX attribute builder for Eiffel web applications. It eliminates tedious string manipulation when generating HTML responses, replacing verbose .append() chains with a clean, type-safe, chainable API.

Part of the simple_* ecosystem of focused, single-purpose Eiffel libraries.

Quick Start

Installation

<library name="simple_htmx" location="$SIMPLE_HTMX\simple_htmx.ecf"/>

Basic Usage

local
    html: HTMX_FACTORY
do
    create html

    -- Create a button with HTMX attributes
    Result := html.button_text ("Save")
        .class_ ("btn btn-primary")
        .hx_post ("/api/save")
        .hx_target ("#result")
        .hx_swap_inner_html
        .to_html_8

    -- Output: <button class="btn btn-primary" hx-post="/api/save"
    --         hx-target="#result" hx-swap="innerHTML">Save</button>
end

Before and After

-- BEFORE (Manual String Building)
Result.append ("<div class=%"control col-")
Result.append (a_control.col_span.out)
Result.append ("%"hx-get=%"/api/data%">")

-- AFTER (With simple_htmx)
Result := html.div
    .class_ ("control")
    .class_ ("col-" + a_control.col_span.out)
    .hx_get ("/api/data")
    .to_html_8
